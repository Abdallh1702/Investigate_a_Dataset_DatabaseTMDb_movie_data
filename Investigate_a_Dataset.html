def calc(colum):
    #for highest earned profit
    high = pd.DataFrame(data.loc[data[colum].idxmax()])
    #for lowest earned profit
    low = pd.DataFrame(data.loc[data[colum].idxmin()])

    informations = pd.concat([high, low], axis=1)

    return informations
